#include <stdio.h>
#include <stdlib.h>
int main(int argc, char *argv[]){    
    int A[3][2]={{4,2},{4,2},{1,0}};
    int B[2][4]={{7,4,1,-1},{3,-2,4,2}};
    int carpim[3][4];    
    int i,j,k,t;    
    printf("\nA Matrisi\n");    
    for(i=0;i<3;i++)    {    
       for(j=0;j<2;j++)    
           printf("%d\t",A[i][j]);    
           printf("\n");                    
       }    
       printf("\nB Matrisi\n");    
       for(i=0;i<2;i++)    {    
          for(j=0;j<4;j++)   
              printf("%d\t",B[i][j]);   
              printf("\n");                    
          }    
          for(k=0;k<4;k++)    {    
              for(i=0;i<3;i++)    {
                  carpim[i][k]=0;    
                  for(j=0;j<2;j++)    {    
                      carpim[i][k]+=A[i][j]*B[j][k];                    
                  }                    
              }                   
          }    
          
          
          printf("\nÝki Matris Çarpýmý\n");      // karmaþýklýðý O(m*n*p) dir
          for(i=0;i<3;i++)    {    
              for(j=0;j<4;j++)  {  
                  printf("%d\t",carpim[i][j]);    
                  printf("\t");                    
              }  
              printf("\n");
          }
          system("PAUSE");	  
          return 0;
}
